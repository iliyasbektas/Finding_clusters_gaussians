Part 1.

Extend my handmade code to get the EM-fit of a generic mixture of k > 1 (instead of just 2) Gaussians.

What is implied here by ‘my handmade code’ is a code contained in this file. In particular, we have to extend ‘handmade.em’ function. This is how original function looks like:

handmade.em <- function(y, p, mu, sigma, n_iter, plot_flag = T)
{
  # Init / 2 components only
  cols     <- c(rgb(1,0,0,.3), rgb(0,1,0,.3))
  like     <- p[1]*dnorm(y, mu[1], sigma[1]) + p[2]*dnorm(y, mu[2], sigma[2])
  deviance <- -2*sum(log(like))
  res      <- matrix(NA,n_iter + 1, 8)
  res[1,]  <- c(0, p, mu, sigma, deviance)
  for (iter in 1:n_iter) {
    # E step
    d1 <- p[1]*dnorm(y, mu[1], sigma[1])
    d2 <- p[2]*dnorm(y, mu[2], sigma[2])
    r1 <- d1/(d1 + d2)
    r2 <- 1 - r1
    
    # M step
    p[1]     <- mean(r1)
    mu[1]    <- sum(r1*y)/sum(r1)
    sigma[1] <-sqrt( sum(r1*(y^2))/sum(r1) - (mu[1])^2 )
    p[2]     <- 1 - p[1]
    mu[2]    <- sum((r2)*y)/sum((r2))
    sigma[2] <- sqrt(sum(r2*(y^2))/sum(r2) - (mu[2])^2)
    
    # -2 x log-likelihood (a.k.a. deviance)
    like     <- p[1]*dnorm(y, mu[1], sigma[1]) + p[2]*dnorm(y, mu[2], sigma[2])
    deviance <- -2*sum( log(like) )
    
    # Save
    res[iter+1,] <- c(iter, p, mu, sigma, deviance)
    
    # Plot
    if (plot_flag){
      hist(y, prob = T, breaks = 30, col = gray(.8), border = NA, 
           main = "", xlab = paste("EM Iteration: ", iter, "/", n_iter, sep = ""))
      set.seed(123)
      points(jitter(y), rep(0,length(y)), 
             pch = 19, cex = .6, 
             col = cols[ (dnorm(y,mu[1],sigma[1]) > dnorm(y,mu[2],sigma[2])) + 1])
      curve(p[1]*dnorm(x,mu[1],sigma[1]) + p[2]*dnorm(x,mu[2],sigma[2]),
            lwd = 4, col = rgb(0,0,0,.5), add = TRUE)
      Sys.sleep(1.5)
    }
  }
  res <- data.frame(res)
  names(res) <- c("iteration","p1","p2","mu1","mu2","sigma1","sigma2","deviance")
  out <- list(parameters = c(p = p, mu = mu, sigma = sigma), deviance = deviance, res = res)
  return(out)
}

To modify original function, we make changes within the original function and have to add some functions for convenience reasons:

Firstly, we should create functions that are going to be implemented.

Let’s start from the easiest one. So, let’s firstly create a function where for each observation we give a color of the Gaussian Model to which it belongs with the highest probability.

likely_color = function(colors, d) {
  a = argmax(d, rows = FALSE) # for finding the class with the largest predicted probability
  return(colors[a]) # the matching color as a result
}

Originally, this function looked like this: ’like = p[1]* dnorm(y, mu[1], sigma[1]) + p[2]*dnorm(y, mu[2], sigma[2])’. We just redo it for multiple p’s. In other words, it calculates for each observation the corresponding likelihood probability.

like_prob = function(y, p, mu, sigma) {
  likelihood = 0
 # for i in range of weights' vector
  for(i in 1:length(p)) {
   
    likelihood = likelihood + p[i] * dnorm(y, mu[i], sigma[i]) 
  }
  
  return(likelihood)
}

Finally, we have to include function that estimates the expected value for each latent variable z.

d1 = p[1]*dnorm(y, mu[1], sigma[1])

d2 = p[2]*dnorm(y, mu[2], sigma[2])

r1 = d1/(d1 + d2)

r2 = 1 - r1

Above is how it used to look like. We are going implement the same logic. We update the probability for each y observation in each of k iterations.

estep = function(y, p, mu, sigma) {
  d = matrix(NA, length(p), length(y)) # define the matrix with k components as number of rows and number of samples as number of columns
  
  for(i in 1:length(p)) {
    d[i,] = p[i] * dnorm(y, mu[i], sigma[i]) # for each k iteration we update the probability for each y observation 
  }
  
  return(d)
}

Final update of the function:

Mainly, corrections are about adjusting the original function to number of Gaussians more than one. Mostly,it is self-explanatory.

handmade.em.new = function(y, p, mu, sigma, n.iter, plot.flag = T)
{
  
  # length of weights' vector = number of Gaussians  =  length(p)
  
  cols = randomColor(count = length(p))
  like = like_prob(y, p, mu, sigma)
  
  deviance = -2 * sum(log(like))
  result = matrix(NA,n.iter + 1, 3 * length(p) + 2) # Previously it was: 'matrix(NA,n_iter + 1, 8)'. This is expected because previously there were 2 Gaussians. The logic here is that the columns are 3*length(p) because 3 are the Gaussian parameters of length(p). In addition, we add two to save  the number of each iteration and the corresponding deviance. 
  result[1,]  = c(0, p, mu, sigma, deviance) 
  
  
  d = estep(y, p, mu, sigma) # implementaion of our 'estep' function
  r = matrix(NA, length(p), length(y)) 
  for (iteration in 1:n.iter) { 
    for(i in 1:length(p)) {
      r[i,] = d[i,] / colSums(d)
    }
    
    # M step - update the theta parameters for the Gaussian components
    for(i in 1:length(p)) {
      p[i] = mean(r[i,])
      mu[i] = sum(r[i,] * y) / sum(r[i,])
      sigma[i] = sqrt(sum(r[i,] * (y^2)) / sum(r[i,]) - (mu[i])^2)
    }
    
    like = like_prob(y, p, mu, sigma) 
    
    # -2 x log-likelihood (a.k.a. deviance)
    deviance = -2 * sum( log(like) )
    
    # Save the results of the i-th iteration
    result[iteration+1, ] = c(iteration, p, mu, sigma, deviance)
    
    # E step - update probabilities matrix
    d = estep(y, p, mu, sigma) 
    
    # Plot
    if (plot.flag){
      hist(y, prob = T, breaks = 50, col = gray(.8), border = NA, 
           main = "", xlab = paste("EM Iteration: ", iteration, "/", n.iter, sep = ""))
      set.seed(123)
      points(jitter(y), rep(0,length(y)), 
             pch = 19, cex = .6, 
             col = likely_color(cols, d)) # we know what this function does
      curve(like_prob(x, p, mu, sigma),
            lwd = 4, col = rgb(0,0,0,.5), add = TRUE)
      Sys.sleep(1.5)
    }
  }
  
  result = data.frame(result)
  names(result) = c("iteration", paste("p", 1:length(p), sep=""), paste("mu", 1:length(p), sep=""), paste("sigma", 1:length(p), sep=""), "deviance")
  out = list(parameters = c(p = p, mu = mu, sigma = sigma), deviance = deviance, res = result) 
  return(out)
}

Part 2.

Pick two different sample sizes, say n1 < n2 to probe a clearly non-asymptotic and a reasonably asymptotic regime.

So, I expect that with the larger size of sample it will look more like a real distribution due to CLT and LLN. With CLT and LLN as we increase size of samples and increase number of samples, it will look more real distribution.

In the function below, after giving equal wieghts to k components, we generate random parameters for mu and sigma.

random_par = function(y, k) {
  return(list(p = rep(1/k, k),sigma = runif(k, min = 0.1, max = 0.9), mu = runif(k, min = 0.1, max= 0.9))) 
}

We are going to use bart distribution and below is its creation:

bart=function(n) {
  return( rnormmix(n,lambda = c(0.5, rep(0.1,5)),mu = c(0, ((0:4)/2)-1),sigma = c(1, rep(0.1,5)) ))}

The function below is necessary for plotting our result for asymptotic and non-asymptotic sample sizes

plotting_result = function(y, p, mu, sigma, n.iter) { 
  cols=randomColor(count = length(p))
  d = estep(y, p, mu, sigma)
    
  hist(y, prob = T, breaks = 50, col = gray(.8), border = NA, 
     main = "", xlab = paste("EM Iteration: ", n.iter, sep = ""))
  set.seed(97)
  points(jitter(y), rep(0,length(y)), 
         pch = 19, cex = .6, 
         col = likely_color(cols, d))
  curve(like_prob(x, p, mu, sigma),
        lwd = 4, col = rgb(0,0,0,.5), add = TRUE)
}
